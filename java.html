Experiment 2: Prime Numbers Between 1 and 1000

Aim: To find and display all prime numbers between 1 and 1000 using a function to check primality.

Source Code

public class PrimeNumbers {
    public static void main(String[] args) {
        System.out.println("Prime numbers between 1 and 1000:");
        for (int num = 2; num <= 1000; num++) {
            if (isPrime(num)) {
                System.out.print(num + " ");
            }
        }
    }
    
    // Function to check if a number is prime
    public static boolean isPrime(int n) {
        if (n < 2) return false;
        for (int i = 2; i <= Math.sqrt(n); i++) {
            if (n % i == 0) {
                return false;
            }
        }
        return true;
    }
}

Experiment 3: Material Calculator for Rectangle

Aim: To calculate the material required for a rectangular notice board (paper for front and back, rubber for edges) using a Shape class.

Source Code

import java.util.Scanner;
public class MaterialCalculator {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        Shape shape = new Shape();
        
        System.out.print("Enter length: ");
        int length = sc.nextInt();
        
        System.out.print("Enter breadth: ");
        int breadth = sc.nextInt();
        int area = shape.areaRectangle(length, breadth);
        int paper = 2 * area;
        // front + back
        int rubber = shape.perimeterRectangle(length, breadth);
        System.out.println("Paper material required = " + paper + " sq.units");
        System.out.println("Rubber material required = " + rubber + " units");
        sc.close();
    }
}
class Shape {
    public int areaRectangle(int l, int b) {
        return l * b;
    }
    
    public int areaSquare(int s) {
        return s * s;
    }
    
    public int perimeterRectangle(int l, int b) {
        return 2 * (l + b);
    }
}

Experiment 4: Shape Area Calculator Using Method Overloading

Aim: To demonstrate method overloading by calculating areas of different shapes (circle, rectangle, triangle, square) using methods with the same name but different parameters.

Source Code

import java.util.Scanner;
public class ShapeAreaApp {
    // Nested helper class for area calculations
    static class AreaCalculator {
        // Area of circle
        public double area(double radius) {
            return Math.PI * radius * radius;
        }
        
        // Area of rectangle
        public double area(double length, double breadth) {
            return length * breadth;
        }
        
        // Area of triangle
        public double area(double base, double height, boolean isTriangle) {
            return 0.5 * base * height;
        }
        
        // Area of square
        public int area(int side) {
            return side * side;
        }
    }
    
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        AreaCalculator calc = new AreaCalculator();
        
        System.out.println("=== Area Calculator ===");
        System.out.println("1. Circle");
        System.out.println("2. Rectangle");
        System.out.println("3. Triangle");
        System.out.println("4. Square");
        System.out.print("Choose a shape (1-4): ");
        
        int choice = sc.nextInt();
        
        switch (choice) {
            case 1:
                System.out.print("Enter radius: ");
                double r = sc.nextDouble();
                System.out.println("Area of Circle = " + calc.area(r));
                break;
            case 2:
                System.out.print("Enter length: ");
                double l = sc.nextDouble();
                System.out.print("Enter breadth: ");
                double b = sc.nextDouble();
                System.out.println("Area of Rectangle = " + calc.area(l, b));
                break;
            case 3:
                System.out.print("Enter base: ");
                double base = sc.nextDouble();
                System.out.print("Enter height: ");
                double h = sc.nextDouble();
                System.out.println("Area of Triangle = " + calc.area(base, h, true));
                break;
            case 4:
                System.out.print("Enter side: ");
                int s = sc.nextInt();
                System.out.println("Area of Square = " + calc.area(s));
                break;
            default:
                System.out.println("Invalid choice!");
        }
        sc.close();
    }
}

Experiment 5: Complex Number Addition Using Constructor Overloading

Aim: To demonstrate constructor overloading by adding two complex numbers where one constructor creates a complex number and another constructor adds two complex numbers.

Source Code

import java.util.Scanner;
public class ComplexCalculator {
    // Nested class for Complex numbers
    static class Complex {
        private double real;
        private double imag;
        
        // Constructor for a single complex number
        public Complex(double r, double i) {
            this.real = r;
            this.imag = i;
        }
        
        // Constructor for addition (takes two Complex objects)
        public Complex(Complex c1, Complex c2) {
            this.real = c1.real + c2.real;
            this.imag = c1.imag + c2.imag;
        }
        
        // Method to display the complex number
        public void display() {
            if (imag >= 0)
                System.out.println(real + "+" + imag + "i");
            else
                System.out.println(real + "-" + Math.abs(imag) + "i");
        }
    }
    
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        // Input first complex number
        System.out.print("Enter real part of first number: ");
        double r1 = sc.nextDouble();
        System.out.print("Enter imaginary part of first number: ");
        double i1 = sc.nextDouble();
        Complex c1 = new Complex(r1, i1);
        
        // Input second complex number
        System.out.print("Enter real part of second number: ");
        double r2 = sc.nextDouble();
        System.out.print("Enter imaginary part of second number: ");
        double i2 = sc.nextDouble();
        Complex c2 = new Complex(r2, i2);
        
        // Add two complex numbers
        Complex sum = new Complex(c1, c2);
        // Display result
        System.out.print("Sum = ");
        sum.display();
        
        sc.close();
    }
}

Experiment 6: Matrix Flip Operations

Aim: To implement matrix manipulation operations including horizontal and vertical flip using a nested class.

Source Code

import java.util.Scanner;
public class MatrixApp {
    // Nested class to manipulate matrices
    static class MatrixManipulator {
        private int[][] matrix;
        private int rows, cols;
        
        // Method to read matrix
        public void readMatrix(Scanner sc) {
            System.out.print("Enter number of rows: ");
            rows = sc.nextInt();
            System.out.print("Enter number of columns: ");
            cols = sc.nextInt();
            
            matrix = new int[rows][cols];
            System.out.println("Enter matrix elements:");
            for (int i = 0; i < rows; i++) {
                for (int j = 0; j < cols; j++) {
                    matrix[i][j] = sc.nextInt();
                }
            }
        }
        
        // Method to display matrix
        public void displayMatrix() {
            System.out.println("Matrix:");
            for (int i = 0; i < rows; i++) {
                for (int j = 0; j < cols; j++) {
                    System.out.print(matrix[i][j] + "\t");
                }
                System.out.println();
            }
        }
        
        // Method to flip matrix horizontally
        public void flipHorizontal() {
            for (int i = 0; i < rows; i++) {
                for (int j = 0, k = cols - 1; j < k; j++, k--) {
        
                    int temp = matrix[i][j];
                    matrix[i][j] = matrix[i][k];
                    matrix[i][k] = temp;
                }
            }
        }
        
        // Method to flip matrix vertically
        public void flipVertical() {
            for (int i = 0, k = rows - 1; i < k; i++, k--) {
                int[] temp 
                    = matrix[i];
                matrix[i] = matrix[k];
                matrix[k] = temp;
            }
        }
    }
    
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        MatrixManipulator mm = new MatrixManipulator();
        
        mm.readMatrix(sc);
        
        System.out.println("\nOriginal Matrix:");
        mm.displayMatrix();
        
        // Flip horizontally
        mm.flipHorizontal();
        System.out.println("\nMatrix after Horizontal Flip:");
        mm.displayMatrix();
        
        // Flip vertically
        mm.flipVertical();
        System.out.println("\nMatrix after Vertical Flip:");
        mm.displayMatrix();
        
        sc.close();
    }
}

Experiment 7: Palindrome String Checker

Aim: To check whether a given string is a palindrome using StringBuffer for string reversal.

Source Code

import java.util.Scanner;
public class Palindrome {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter a string: ");
        String str = sc.nextLine();
        
        StringBuffer buffer = new StringBuffer(str);
        String rev = buffer.reverse().toString();
        if (str.equalsIgnoreCase(rev)) {
            System.out.println("The string is a palindrome.");
        } else {
            System.out.println("The string is not a palindrome.");
        }
        
        sc.close();
    }
}

Experiment 8: Sphere Volume Calculator Using Multilevel Inheritance

import java.util.Scanner;
class Area extends Circle {
    double area;
    void calculate() {
        area = Math.PI * radius * radius;
    }
    
    void display() {
        System.out.println("Area of Circle = " + area);
    }
}

    double volume;
    void calculate() {
        super.calculate();
        volume = (4.0 / 3.0) * Math.PI * radius * radius * radius;

    void display() {
        System.out.println("Volume of Sphere = " + volume);
    }
}
public class Circle {
    double radius;
    
    void getRadius() {
        Scanner sc = new Scanner(System.in);
        System.out.print("Enter radius: ");
        radius = sc.nextDouble();
        sc.close();
    }
    
    public static void main(String[] args) {
        Volume vol = new Volume();
        vol.getRadius();
        vol.calculate();
        vol.display();
    }
}

Experiment 10: Custom Exception Handling

Aim: To create and use a custom exception (`OddNumberException`) to handle invalid input in a method.

class OddNumberException extends Exception {
    public OddNumberException(String message) {
        super(message);
    }
}
public class EvenNumberSquare {
    public static void calculateSquare(int number) throws OddNumberException {
        if (number % 2 != 0) {
            throw new OddNumberException("Input number must be an even number: " + number);
        }
        int square = number * number;
        System.out.println("Square of " + number + " is: " + square);
    }
    public static void main(String[] args) {
        try {
            // Test with even number
            calculateSquare(4);
            // Test with odd number
            calculateSquare(7);
        } catch (OddNumberException e) {
            System.out.println("Error: " + e.getMessage());
        }
    }
}

Experiment 11: Multithreading with Runnable

Source Code

import java.util.Scanner;
import java.io.File;
class Numbers implements Runnable {
    public void run() {
        for (int i=1; i<=10; i++) {
            System.out.println("Number: " + i);
            try {
                Thread.sleep(100);
            } catch (InterruptedException e) {
                System.out.println(e);
            }
        }
    }
}
class Squares implements Runnable {
    public void run() {
        for (int i=1; i<=10; i++) {
            System.out.println("Square: " + (i * i));
            try {
                Thread.sleep(100);
            } catch (InterruptedException e) {
                System.out.println(e);
            }
        }
    }
}
public class Piy_11{
    public static void main(String[] args) {
        Thread t1= new Thread(new Numbers());
        Thread t2= new Thread(new Squares());
        t1.start();
        t2.start();
    }
}
